Need to Use CORS Middleware on NodeJS Side to enable front-end from other port to use that nodeJS services
Lets Start Some Coding
AngularJS
var express = require('express');
var cors=require('cors');
var app = express();

app.use(cors());
app.get('/', function(req, res) {

	res.setHeader('Content-type','application/json');

	var json={

		name:'Ajay Gaddam'
	};


    res.json(json);
});
app.get('/google', function(req, res) {
    res.send('Hello Google');
});
app.listen(3000);
NodeJS


var app = angular.module("first_module", []);
app.controller("first_controller", function($scope, $http) {
    $scope.name = "Ajay";
    $http.get('http://localhost:3000/')
        .then(function(response) {
            $scope.name = response.data.name;
        })
});
